```powershell
# 1. Create topic with 3 partitions
docker exec kafka kafka-topics --create \
  --topic multi-topic \
  --bootstrap-server localhost:9092 \
  --partitions 3 \
  --replication-factor 1

# 2. Run 3 consumers (in separate terminals)
.\kafka-demo.exe -mode consumer -topic multi-topic -group my-group
.\kafka-demo.exe -mode consumer -topic multi-topic -group my-group
.\kafka-demo.exe -mode consumer -topic multi-topic -group my-group

# 3. Run producer
.\kafka-demo.exe -mode producer -topic multi-topic

# Result: Messages distributed across 3 partitions and 3 consumers!
```
